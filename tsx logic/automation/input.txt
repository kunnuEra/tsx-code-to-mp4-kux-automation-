    extrapolateLeft: 'clamp',
    extrapolateRight: 'clamp',

    extrapolateLeft: 'clamp',
    extrapolateRight: 'clamp',

    extrapolateRight: 'clamp',

    extrapolateRight: 'clamp',

        />

              fill="none"
              stroke="#9E9E9E"
              strokeWidth="2"
            />
            <path
              d={`M0,600 Q480,${680 + droopY * 0.7} 960,${650 + droopY * 0.3} Q1440,${620 - droopY * 0.5} 1920,680`}

import React from 'react';
import {

    extrapolateLeft: 'clamp',
    extrapolateRight: 'clamp',

    extrapolateLeft: 'clamp',
    extrapolateRight: 'clamp',

  const particles = Array.from({ length: 25 }, (_, i) => {

    extrapolateLeft: 'clamp',
    extrapolateRight: 'clamp',

    extrapolateLeft: 'clamp',
    extrapolateRight: 'clamp',

        />

            position: 'absolute',
            left: 80,
            top: 120,

            height: 280,

            opacity: 0.04,
            transform: `rotate(${shapeRot}deg)`,

                position: 'absolute',
                left: 50 + i * 65,

                height: '100%',
                background: '#9E9E9E',

            />

                  {up ? '▲' : '▼'} {(seededRandom(i + 5800) * 3).toFixed(1)}%
                </span>

            })}
          </div>
        </div>

        {/* Candlestick chart */}
        <svg
          width="1800"
          height="500"
          viewBox="0 0 1800 500"
          style={{
            position: 'absolute',
            left: '2%',
            top: '12%',
            opacity: interpolate(frame, [25, 50], [0, 0.6], {
              extrapolateLeft: 'clamp',
              extrapolateRight: 'clamp',

        >
          {/* Grid lines */}
          {Array.from({ length: 6 }, (_, i) => (
