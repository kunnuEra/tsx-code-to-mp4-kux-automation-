            opacity: interpolate(frame, [30, 80], [0, 0.08], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' }),
            transform: `rotate(${frame * 0.3}deg)`,
          }}
          viewBox="0 0 300 300"
        >
          {[...Array(6)].map((_, i) => (
            <circle
              key={i}
              cx="150"
              cy="150"
              r={30 + i * 25}
              fill="none"
              stroke="#e9d700"
              strokeWidth="1"
              strokeDasharray={`${5 + i * 3} ${10 + i * 4}`}
            />
          ))}
        </svg>

        {/* Right side - Heavy bill / weight */}
        <div
          style={{
            position: 'absolute',
            right: 100,
            top: 180,
            textAlign: 'right',
            width: 650,
          }}
        >
          <div
            style={{

              color: '#F5F5F5',

          </div>

              opacity: sp(20),

            बाद में आता है

          <div
            style={{
              marginTop: 50,
              opacity: sp(impactFrame - 20),
              transform: `translateY(${interpolate(
                frame,
                [impactFrame - 30, impactFrame],

              {/* Chain links above */}
              <ellipse cx="100" cy="10" rx="15" ry="12" fill="none" stroke="#9E9E9E" strokeWidth="2" />

          </div>

          {/* Crack lines on impact */}
          {isImpact && (
            <svg width="300" height="50" style={{ position: 'absolute', right: 20, bottom: 180, opacity: 0.5 }}>

        </div>

        {/* Bottom - quote bar */}
        <div
          style={{
            position: 'absolute',
            bottom: 80,
            left: 0,
            width: '100%',
            textAlign: 'center',
            opacity: sp(120),
            transform: `translateY(${interpolate(sp(120), [0, 1], [30, 0])}px)`,
          }}
        >
          <div
            style={{
              display: 'inline-block',
              padding: '14px 45px',

              borderRadius: 12,
              border: '1px solid rgba(255,0,0,0.2)',

          >
            <span style={{ fontFamily: 'Noto Sans Devanagari', fontSize: 40, color: '#F5F5F5' }}>

        {/* Stick figure being pulled by megaphone sound */}
        <div
          style={{
            position: 'absolute',
            left: 500,
            top: 450,
            opacity: sp(40),
            transform: `translateX(${-10 + Math.sin(frame * 0.04) * 8}px) translateY(${Math.sin(frame * 0.05) * 2}px)`,
          }}
        >
          <svg width="100" height="160" viewBox="0 0 100 160">
            <circle cx="50" cy="28" r="28" fill="rgba(233,215,0,0.04)" />
            <circle cx="50" cy="28" r="15" fill="none" stroke="#F5F5F5" strokeWidth="2" />
            <circle cx="43" cy="26" r="5.5" fill="none" stroke="#FFC107" strokeWidth="1.5" />
            <circle cx="57" cy="26" r="5.5" fill="none" stroke="#FFC107" strokeWidth="1.5" />
            <line x1="48" y1="26" x2="52" y2="26" stroke="#FFC107" strokeWidth="1.2" />
            {/* Dazed expression - spiral eyes */}
            <circle cx="43" cy="26" r="3" fill="none" stroke="#e9d700" strokeWidth="1" opacity={0.5 + Math.sin(frame * 0.1) * 0.3} />
            <line x1="50" y1="43" x2="50" y2="100" stroke="#F5F5F5" strokeWidth="2" />
            {/* Leaning towards left (sound) */}
            <line x1="50" y1="65" x2="25" y2="50" stroke="#F5F5F5" strokeWidth="2" />
            <line x1="50" y1="65" x2="75" y2="85" stroke="#F5F5F5" strokeWidth="2" />
            <line x1="50" y1="100" x2="35" y2="145" stroke="#F5F5F5" strokeWidth="2" />
            <line x1="50" y1="100" x2="65" y2="145" stroke="#F5F5F5" strokeWidth="2" />
          </svg>
        </div>
      </AbsoluteFill>

      {/* Vignette */}
      <AbsoluteFill
        style={{
          background: 'radial-gradient(ellipse at center, transparent 35%, rgba(0,0,0,0.75) 100%)',
          pointerEvents: 'none',
        }}
      />

      {/* Watermark */}
      <div
        style={{
          position: 'absolute',
          bottom: 30,
          right: 45,
          fontFamily: 'Inter',
          fontSize: 22,
          fontWeight: 600,
          color: '#9E9E9E',
          opacity: interpolate(frame, [0, 30], [0, 0.3], { extrapolateRight: 'clamp' }),
        }}
      >
        ProBuffet HQ
      </div>
    </AbsoluteFill>
  );
};
