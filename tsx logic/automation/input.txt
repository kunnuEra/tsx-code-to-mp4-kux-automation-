            opacity: redFlash,
          }}
        />

        {/* Shape layers — zigzag */}
        <AbsoluteFill style={{ opacity: 0.03 }}>
          <svg width="1920" height="1080" viewBox="0 0 1920 1080">
            <path
              d={`M0,540 ${Array.from({ length: 20 }, (_, i) => `L${i * 100 + 50},${540 + (i % 2 === 0 ? -30 : 30) + Math.sin(zigOffset + i) * 5} L${(i + 1) * 100},540`).join(' ')}`}
              fill="none"
              stroke="#ff0000"
              strokeWidth="1"
            />
          </svg>
        </AbsoluteFill>

        {/* Shape layers — broken circle */}
        <div
          style={{
            position: 'absolute',
            left: '50%',
            top: '32%',
            width: 400,
            height: 400,
            transform: 'translate(-50%, -50%)',
            opacity: 0.05,
          }}
        >
          <svg viewBox="0 0 400 400">
            <circle
              cx="200"
              cy="200"
              r="180"
              fill="none"
              stroke="#ff0000"
              strokeWidth="2"
              strokeDasharray="30 20"
              transform={`rotate(${frame * 0.08} 200 200)`}
            />
          </svg>
        </div>

        {/* Particles */}
        {particles.map((p, i) => (
          <div
            key={`p18-${i}`}
            style={{
              position: 'absolute',
              left: p.x,
              top: p.y + p.dy,
              width: p.size,
              height: p.size,
              borderRadius: '50%',
              backgroundColor: i % 2 === 0 ? '#ff0000' : '#9E9E9E',
              opacity: p.opacity,
            }}
          />
        ))}

        {/* "इतनी सी बात" — fading out */}
        <div
          style={{
            position: 'absolute',
            top: '15%',
            left: '50%',
            transform: `translate(-50%, 0) scale(${simpleScale})`,
            opacity: simpleOp * interpolate(frame, [60, 90], [1, 0.2], {
              extrapolateLeft: 'clamp',
              extrapolateRight: 'clamp',
            }),
            textAlign: 'center',
          }}
        >
          <div
            style={{
              fontFamily: "'Noto Sans Devanagari', sans-serif",
              fontSize: 50,
              fontWeight: 600,
              color: '#9E9E9E',
            }}
          >
            बस आवंटन सही था,
          </div>
          <div
            style={{
              fontFamily: "'Noto Sans Devanagari', sans-serif",
              fontSize: 60,
              fontWeight: 800,
              color: '#F5F5F5',
            }}
          >
            इतनी सी बात.
          </div>
        </div>

        {/* Crack lines */}
        <svg
          style={{
            position: 'absolute',
            left: 0,
            top: 0,
            width: 1920,
            height: 1080,
            pointerEvents: 'none',
          }}
        >
          {cracks.map((crack, i) => (
            <line
              key={`crack-${i}`}
              x1={crack.x1}
              y1={crack.y1}
              x2={crack.x2}
              y2={crack.y2}
              stroke="#ff0000"
              strokeWidth="2"
              opacity={0.4 * crackProgress}
            />
          ))}
        </svg>

        {/* "गलतफ़हमी" text — shattered through */}
        <div
          style={{
            position: 'absolute',
            left: '50%',
            top: '35%',
            transform: `translate(-50%, -50%) scale(${galScale})`,
            opacity: galOp,
            textAlign: 'center',
          }}
        >
          {/* Warning triangle */}
          <div style={{ display: 'inline-block', marginBottom: 15, transform: `scale(${warnPulse})` }}>
            <svg width="80" height="70" viewBox="0 0 80 70">
              <polygon points="40,5 75,65 5,65" fill="none" stroke="#ff0000" strokeWidth="3" />
              <text x="40" y="55" textAnchor="middle" fill="#ff0000" fontSize="30" fontFamily="Inter" fontWeight="900">!</text>
            </svg>
          </div>
          <div
            style={{
              fontFamily: "'Yantra One', sans-serif",
              fontSize: 120,
              fontWeight: 900,
              color: '#ff0000',
              textShadow: '0 0 60px rgba(255,0,0,0.4), 0 0 120px rgba(255,0,0,0.2)',
              lineHeight: 1,
            }}
          >
            गलतफ़हमी
          </div>
          <div
            style={{
              fontFamily: "'Noto Sans Devanagari', sans-serif",
              fontSize: 34,
              fontWeight: 500,
              color: '#9E9E9E',
              marginTop: 12,
            }}
          >
            जो शायद तुम्हारे भीतर भी बैठी हो
          </div>
        </div>

        {/* "महंगी पड़ती है" impact */}
        <div
          style={{
            position: 'absolute',
            bottom: '15%',
            left: '50%',
            transform: `translate(-50%, 0) scale(${costScale})`,
            opacity: costOp,
            textAlign: 'center',
          }}
        >
          <div
            style={{
              fontFamily: "'Noto Sans Devanagari', sans-serif",
              fontSize: 48,
              fontWeight: 700,
              color: '#F5F5F5',
            }}
          >
            और ये गलतफ़हमी
          </div>
          <div
            style={{
              fontFamily: "'Yantra One', sans-serif",
              fontSize: 90,
              fontWeight: 900,
              color: '#ff0000',
              textShadow: '0 0 50px rgba(255,0,0,0.4)',
            }}
          >
            बहुत महंगी पड़ती है
          </div>
        </div>

        {/* Tease — "क्या है वो?" */}
        <div
          style={{
            position: 'absolute',
            bottom: '4%',
            left: '50%',
            transform: 'translate(-50%, 0)',
            opacity: teaseOp,
          }}
        >
          <div
            style={{
              fontFamily: "'Noto Sans Devanagari', sans-serif",
              fontSize: 30,
              fontWeight: 600,
              color: '#FFC107',
              animation: 'pulse 1.5s infinite',
            }}
          >
            क्या है वो गलतफ़हमी? ↓
          </div>
        </div>

        {/* Stick figure — worried */}
        <div
          style={{
            position: 'absolute',
            right: '8%',
            top: '45%',
            transform: `scale(${sp(70)}) translateY(${Math.sin(frame * 0.05) * 2}px)`,
            opacity: interpolate(frame, [70, 90], [0, 0.8], {
              extrapolateLeft: 'clamp',
              extrapolateRight: 'clamp',
            }),
          }}
        >
          <svg width="65" height="100" viewBox="0 0 65 100">
            <circle cx="32" cy="16" r="20" fill="rgba(255,0,0,0.06)" />
            <circle cx="32" cy="16" r="12" fill="none" stroke="#F5F5F5" strokeWidth="2.5" />
            <circle cx="28" cy="14" r="3.5" fill="none" stroke="#FFC107" strokeWidth="1.2" />
            <circle cx="36" cy="14" r="3.5" fill="none" stroke="#FFC107" strokeWidth="1.2" />
            <line x1="31" y1="14" x2="33" y2="14" stroke="#FFC107" strokeWidth="0.8" />
            {/* Worried mouth */}
            <path d="M27,22 Q32,19 37,22" fill="none" stroke="#F5F5F5" strokeWidth="1.5" />
            <line x1="32" y1="28" x2="32" y2="62" stroke="#F5F5F5" strokeWidth="2.5" strokeLinecap="round" />
            {/* Hands on head */}
            <line x1="32" y1="38" x2="18" y2="22" stroke="#F5F5F5" strokeWidth="2.5" strokeLinecap="round" />
            <line x1="32" y1="38" x2="46" y2="22" stroke="#F5F5F5" strokeWidth="2.5" strokeLinecap="round" />
            <line x1="32" y1="62" x2="22" y2="90" stroke="#F5F5F5" strokeWidth="2.5" strokeLinecap="round" />
            <line x1="32" y1="62" x2="42" y2="90" stroke="#F5F5F5" strokeWidth="2.5" strokeLinecap="round" />
          </svg>
        </div>

        {/* Vignette — heavy */}
        <AbsoluteFill
          style={{
            background:
              'radial-gradient(ellipse at center, transparent 35%, rgba(0,0,0,0.75) 100%)',
            pointerEvents: 'none',
          }}
        />

        {/* Watermark */}
        <div
          style={{
            position: 'absolute',
            bottom: 30,
            right: 45,
            fontFamily: "'Inter', sans-serif",
            fontSize: 22,
            fontWeight: 600,
            color: '#9E9E9E',
            opacity: 0.3,
          }}
        >
          ProBuffet HQ
        </div>
      </div>
    </AbsoluteFill>
  );
};

export default Scene18_CostlyMisbelief;
